function arbitrage(z,k,i,Z,K,I,p)
    (;ρ, δ, α, β, γ) = p

    y = exp(z)*k^α # lower case for today
    c = y-i
    Y = exp(Z)*K^α # upper case for tomorrow
    C = Y-I

    r = β*(C/c)^(-γ)*( (1-δ) + α*Y/K ) - 1

    return (r, ) # tuple with one value

end
